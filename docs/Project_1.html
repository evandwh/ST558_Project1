interval_midpoint <- function(time_string) {
  
  if (time_string == "0") {
    return(hms::as_hms(NA))
  }

  pieces <- str_split(time_string, " to ", simplify = TRUE)

  start <- str_trim(pieces[1])
  end <- str_trim(pieces[2])
  
  start <- str_replace_all(start, " a.m.", " AM")
  start <- str_replace_all(start, " p.m.", " PM")
  
  end <- str_replace_all(end, " a.m.", " AM")
  end <- str_replace_all(end, " p.m.", " PM")
  
  start_time <- parse_date_time(start, orders = "I:M p")
  end_time <- parse_date_time(end, orders = "I:M p")

  midpoint <- start_time + (end_time - start_time) / 2
  hms::as_hms(midpoint)
}

census_rawdata_to_tibble <- function(raw_data) {
  numeric_cols <- c("AGEP", "GASP", "GRPIP", "JWMNP", "PWGTP")
  parsed <- fromJSON(rawToChar(raw_data$content))
  data <- as_tibble(parsed[-1, ])
  colnames(data) <- parsed[1, ]
  if ("JWAP" %in% colnames(data)) {
    data <- data |> 
      mutate(
        JWAP = jwap_time_lookup[JWAP]
      )
  }
  if ("JWDP" %in% colnames(data)) {
    data <- data |> 
      mutate(
        JWDP = jwdp_time_lookup[JWDP]
      )
  }
  data <- data |>
    mutate(across(any_of(numeric_cols), as.numeric)) |>
    mutate(
      JWAP = na_if(as.character(JWAP), "0"),
      JWDP = na_if(as.character(JWDP), "0"),
      JWAP = parse_hm(JWAP),
      JWDP = parse_hm(JWDP)
    )
  if ("SEX" %in% colnames(data)) {
      data$SEX = factor(data$SEX, 
                        levels = c(1, 2),
                        labels = c("Male", "Female"))
        }
  if ("FER" %in% colnames(data)) {
      data$FER = factor(data$FER, 
                        levels = c(0, 1, 2),
                        labels = c("N/A (less than 15 years/greater than 50 years/ male)",
                                   "Yes",

  if ("HHL" %in% colnames(data)) {

                                      "Asian and Pacific Island languages",
                                      "Other Language"))

  if ("SCL" %in% colnames(data)) {      
      data$SCL = factor(data$SCL, 
                           levels = c(0,1,2,3),

  )

    } else if (geo_name %in% state_fips) {
      code <- geo_name
    } else {
      stop("State code not found")
    }
  } else if (geo == "REGION") {
    if (geo_name %in% names(region_codes)) {
      code <- region_codes[[geo_name]]
    } else if (geo_name %in% region_codes) {
      code <- geo_name
    } else {
      stop("Region code not found")
    }

    } else if (geo_name %in% division_codes) {
      code <- geo_name

single_year_rawdata <- function(year = 2022, 
                       num_vars = c("AGEP", "PWGTP"),  #AGEP and PWGTP by default
                       cat_vars = c("SEX"), #SEX by default
                       geo = "STATE",       #State by default
                       geo_name = "NC",
                       key = "70c77fa9cc244f5a9fb3c46ed731835d132c0670") {
  num_vars <- toupper(num_vars)
  cat_vars <- toupper(cat_vars)
  geo <- toupper(geo)
  if (year > 2022 || year < 2010) {
    stop("Year must be between 2010 and 2022, inclusive.")
  }
  if (!("PWGTP" %in% num_vars)) {
    num_vars <- c(num_vars, "PWGTP")  #Always want PWGTP
  }
  if (length(num_vars) <= 1) {
    stop("Please provide more than one numerical variable.")
  }
  for (k in num_vars) {
    if (!(k %in% c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP", "PWGTP"))) {
      stop("Please choose only valid options for num_vars.")
    }
  }
  for (k in cat_vars) {
    if (!(k %in% c("FER", "HHL", "SCH", "SCHL", "SEX"))) {
      stop("Please choose only valid options for cat_vars.")
    }
  }
  if (length(cat_vars) == 0) {  #Must have at least one categorical variable.
    stop("Must have at least one categorical variable.")
  }
  if (!(is.null(geo))) {
    if (!(toupper(geo) %in% c("REGION", "DIVISION", "STATE"))) {
      stop("Incorrect Geography provided")
    }
    geo_string <- paste0("&for=",geo)
  } else {
    geo_string <- ""
  }
  if ((geo_string == "") && (!(is.null(geo_name)))) {
    stop("Cannot give a location without a specificed geography type")
  }
  if (geo_name == "ALL") {
    code <- "*"
  }
  else {
    code <- help_codes(geo, geo_name)
  }
  geo_string <- paste0(geo_string,":",code)
  num_string <- paste(num_vars, collapse = ",")
  cat_string <- paste(cat_vars, collapse = ",")
  url <- paste0("https://api.census.gov/data/",
                year,
                "/acs/acs1/pums?get=",
                cat_string,
                ",", #cat_var and num_var always have at least one element
                num_string,
                geo_string,
                "&key=",
                key)
  print(url)
  results <- GET(url)
}

weighted_mean <- function(col, weight) {
  sum(col * weight, na.rm = TRUE) / sum(weight, na.rm = TRUE)
}

weighted_sd <- function(col, weight) {
  mu <- weighted_mean(col, weight)
  sqrt(
    sum(col^2 * weight, na.rm = TRUE) / sum(weight, na.rm = TRUE) - mu^2
  )
}
summary.census <- function(data, num_vars = NULL, cat_vars = NULL) {
  if (is.null(num_vars)) {
    num_vars <- data |> 
      select(where(is.numeric)) |> 
      names()
    num_vars <- setdiff(num_vars, c("year","PWGTP"))
  }
  if (is.null(cat_vars)) {
    cat_vars <- data |>
      select(where(is.factor)) |>
      names()
  }
  numerical_summary <- data |> 
    summarise(
      across(
        all_of(num_vars), 
        list(
          mean = ~ weighted_mean(.x, PWGTP),
          sd   = ~ weighted_sd(.x, PWGTP)
          ),
        .names = "{.col}_{.fn}") |>
    pivot_longer(
      everything(),
      names_to = c("variable", "statistic"),
      names_sep = "_",
      values_to = "value"
    )
    )
  
  categorical_summary <- data |>
    pivot_longer(
      cols = all_of(cat_vars),
      names_to = "variable",
      values_to = "level"
    ) |>
    count(variable, level)
  
  full_summary <- list(
    "Numeric_Summary" = numerical_summary, 
    "Categorical_Summary" = categorical_summary)
  full_summary
}
