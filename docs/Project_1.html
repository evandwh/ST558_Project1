helper_fun <- function(year = 2022, 
                       num_vars = c("AGEP", "PWGTP"),  #AGEP and PWGTP by default
                       cat_vars = c("SEX"), #SEX by default
                       geo = "STATE",       #State by default
                       geo_name = "NC") {   #North Carolina by default
  if (year > 2022 || year < 2010) {
    stop("Year must be between 2010 and 2022, inclusive.")
  }
  if (!("PWGTP" %in% num_vars)) {
    num_vars <- c(num_vars, "PWGTP")  #Always want PWGTP
  }
  for (k in num_vars) {
    if (!(k %in% c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP", "PWGTP"))) {
      stop("Please choose only valid options for num_vars.")
    }
  }
  for (k in cat_vars) {
    if (!(k %in% c("FER", "HHL", "SCH", "SCHL", "SEX"))) {
      stop("Please choose only valid options for cat_vars.")
    }
  }
  if (length(cat_vars) == 0) {  #Must have at least one categorical variable.
    stop("Must have at least one categorical variable.")
  }
  if (!(geo %in% c("REGION", "DIVISION", "STATE"))) {
    stop("Incorrect Geography provided")
  }
  if (geo_name == "ALL") {
    code <- ""
  }
  else {
    code <- help_help_codes(geo_name)
  }
  
  print(year)
  print(num_vars)
  print(cat_vars)
  print(geo)
  print(code)
}
